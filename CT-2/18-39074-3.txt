CREATE TABLE Trainer (
    t_id NUMBER(10) PRIMARY KEY,
    t_name VARCHAR2(50),
    t_address VARCHAR2(100),
    salary NUMBER(10, 2),
    specialty VARCHAR2(50)
);

CREATE TABLE Member (
    m_id NUMBER(10) PRIMARY KEY,
    m_name VARCHAR2(50),
    m_address VARCHAR2(100),
    bill NUMBER(10, 2),
    t_id NUMBER(10),
    CONSTRAINT fk_trainer FOREIGN KEY (t_id) REFERENCES Trainer(t_id)
);
