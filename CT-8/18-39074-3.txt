Left Outer Join:
SELECT borrower.customer_name, loan.loan_number, loan.branch_name, loan.amount
FROM borrower
LEFT JOIN loan ON borrower.loan_number = loan.loan_number;
----------------------------------------------------------------------------------------
Right Outer Join:
SELECT borrower.customer_name, loan.loan_number, loan.branch_name, loan.amount
FROM borrower
RIGHT JOIN loan ON borrower.loan_number = loan.loan_number;
----------------------------------------------------------------------------------------
Full Outer Join:
SELECT borrower.customer_name, loan.loan_number, loan.branch_name, loan.amount
FROM borrower
FULL OUTER JOIN loan ON borrower.loan_number = loan.loan_number;
